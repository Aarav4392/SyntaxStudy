function clamp(value, min, max) {
  return Math.min(max, Math.max(min, value));
}

function daysFromNow(days) {
  return new Date(Date.now() + days * 86400000).toISOString();
}

function loadCards() {
  try {
    const saved = localStorage.getItem("superquiz.cards");
    return saved ? JSON.parse(saved) : seedCards;
  } catch {
    return seedCards;
  }
}

function loadHistory() {
  try {
    const saved = localStorage.getItem("superquiz.history");
    return saved
      ? JSON.parse(saved)
      : [
          { day: "Mon", score: 52 },
          { day: "Tue", score: 57 },
          { day: "Wed", score: 61 },
          { day: "Thu", score: 66 },
          { day: "Fri", score: 69 },
          { day: "Sat", score: 73 },
          { day: "Sun", score: 78 },
        ];
  } catch {
    return [];
  }
}

function AppShell() {
  const [cards, setCards] = useState(loadCards);
  const [history, setHistory] = useState(loadHistory);
  const [activeDeck, setActiveDeck] = useState("all");
  const [mode, setMode] = useState("dashboard");
  const [query, setQuery] = useState("");
  const [dark, setDark] = useState(false);
  const [streak, setStreak] = useState(Number(localStorage.getItem("superquiz.streak") || 4));

  useEffect(() => {
    localStorage.setItem("superquiz.cards", JSON.stringify(cards));
  }, [cards]);

  useEffect(() => {
    localStorage.setItem("superquiz.history", JSON.stringify(history));
  }, [history]);

  useEffect(() => {
    localStorage.setItem("superquiz.streak", String(streak));
  }, [streak]);

  const decks = useMemo(() => {
    const subjects = Array.from(new Set(cards.map((c) => c.subject))).sort();
    const emojis = ["🧠", "⚡", "🚀", "🎯", "🔬", "📝", "🌎", "💡"];
    return [defaultDecks[0], ...subjects.map((s, i) => ({ id: s, name: s, emoji: emojis[i % emojis.length] }))];
  }, [cards]);

  const filteredCards = useMemo(() => {
    return cards.filter((card) => {
      const inDeck = activeDeck === "all" || card.subject === activeDeck;
      const inQuery = `${card.term} ${card.answer} ${card.subject}`.toLowerCase().includes(query.toLowerCase());
      return inDeck && inQuery;
    });
  }, [cards, activeDeck, query]);

  const dueCards = useMemo(() => filteredCards.filter((c) => new Date(c.nextReview) <= new Date()), [filteredCards]);
  const weakCards = useMemo(() => [...filteredCards].sort((a, b) => a.mastery - b.mastery).slice(0, 5), [filteredCards]);
  const avgMastery = filteredCards.length ? Math.round(filteredCards.reduce((s, c) => s + c.mastery, 0) / filteredCards.length) : 0;

  function addCard(card) {
    setCards((prev) => [{ ...card, id: crypto.randomUUID(), mastery: 10, nextReview: new Date().toISOString(), lastResult: "New" }, ...prev]);
  }

  function deleteCard(id) {
    setCards((prev) => prev.filter((c) => c.id !== id));
  }

  function updateCardResult(id, quality) {
    setCards((prev) =>
      prev.map((card) => {
        if (card.id !== id) return card;
        const delta = quality === "again" ? -14 : quality === "hard" ? 5 : quality === "good" ? 12 : 22;
        const days = quality === "again" ? 0 : quality === "hard" ? 1 : quality === "good" ? Math.max(2, Math.round(card.mastery / 25)) : Math.max(4, Math.round(card.mastery / 16));
        return {
          ...card,
          mastery: clamp(card.mastery + delta, 0, 100),
          difficulty: clamp(card.difficulty + (quality === "again" ? 1 : quality === "easy" ? -1 : 0), 1, 5),
          nextReview: daysFromNow(days),
          lastResult: quality === "again" ? "Needs review" : quality === "hard" ? "Hard" : quality === "good" ? "Good" : "Mastered",
        };
      })
    );
    setHistory((prev) => {
      const next = [...prev.slice(-6), { day: new Date().toLocaleDateString(undefined, { weekday: "short" }), score: clamp(avgMastery + 2, 0, 100) }];
      return next;
    });
  }

  function importFromNotes(text, subject) {
    const sentences = text
      .split(/[.?!\n]/)
      .map((s) => s.trim())
      .filter((s) => s.length > 20)
      .slice(0, 8);

    const generated = sentences.map((sentence) => {
      const words = sentence.split(" ");
      const term = words.slice(0, Math.min(5, words.length)).join(" ").replace(/[:,;]/g, "");
      return {
        id: crypto.randomUUID(),
        term: term.length > 8 ? term : `Key idea ${Math.floor(Math.random() * 99)}`,
        answer: sentence,
        subject: subject || "Imported Notes",
        difficulty: 3,
        mastery: 5,
        nextReview: new Date().toISOString(),
        lastResult: "AI-generated",
        sourceNote: sentence,
      };
    });

    if (generated.length) setCards((prev) => [...generated, ...prev]);
  }

  return (
    <div className={dark ? "dark" : ""}>
      <div className="min-h-screen bg-slate-50 text-slate-950 transition-colors dark:bg-slate-950 dark:text-slate-50">
        <div className="mx-auto flex max-w-7xl gap-4 p-4 md:p-6">
          <aside className="hidden w-72 shrink-0 rounded-3xl border border-slate-200 bg-white/80 p-4 shadow-sm backdrop-blur dark:border-slate-800 dark:bg-slate-900/70 lg:block">
            <div className="mb-8 flex items-center gap-3">
              <div className="rounded-2xl bg-slate-950 p-3 text-white dark:bg-white dark:text-slate-950">
                <Brain className="h-6 w-6" />
              </div>
              <div>
                <h1 className="text-xl font-black tracking-tight">SuperQuiz</h1>
                <p className="text-xs text-slate-500">Beyond flashcards</p>
              </div>
            </div>

            <nav className="space-y-2">
              {[
                ["dashboard", LineChart, "Dashboard"],
                ["study", BookOpen, "Smart Study"],
                ["test", GraduationCap, "Test Generator"],
                ["tutor", MessageCircle, "AI Tutor"],
                ["notes", Wand2, "Notes → Cards"],
                ["focus", Timer, "Focus Room"],
                ["library", ListChecks, "Card Library"],
              ].map(([id, Icon, label]) => (
                <button
                  key={id}
                  onClick={() => setMode(id)}
                  className={`flex w-full items-center gap-3 rounded-2xl px-4 py-3 text-left text-sm font-semibold transition ${
                    mode === id
                      ? "bg-slate-950 text-white shadow-lg shadow-slate-200 dark:bg-white dark:text-slate-950 dark:shadow-none"
                      : "text-slate-600 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-800"
                  }`}
                >
                  <Icon className="h-4 w-4" />
                  {label}
                </button>
              ))}
            </nav>

            <div className="mt-8 rounded-3xl bg-gradient-to-br from-slate-900 to-slate-700 p-4 text-white dark:from-slate-100 dark:to-white dark:text-slate-950">
              <div className="mb-3 flex items-center gap-2 text-sm font-bold">
                <Sparkles className="h-4 w-4" /> Study Coach
              </div>
              <p className="text-sm opacity-90">{tips[(streak + filteredCards.length) % tips.length]}</p>
            </div>
          </aside>

          <main className="min-w-0 flex-1">
            <header className="mb-4 flex flex-col gap-3 rounded-3xl border border-slate-200 bg-white/80 p-4 shadow-sm backdrop-blur dark:border-slate-800 dark:bg-slate-900/70 md:flex-row md:items-center md:justify-between">
              <div>
                <p className="text-sm font-semibold text-slate-500">Welcome back</p>
                <h2 className="text-2xl font-black tracking-tight md:text-3xl">Study smarter, not longer.</h2>
              </div>
              <div className="flex flex-wrap items-center gap-2">
                <div className="relative">
                  <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-slate-400" />
                  <input
                    value={query}
                    onChange={(e) => setQuery(e.target.value)}
                    placeholder="Search cards..."
                    className="w-52 rounded-2xl border border-slate-200 bg-white py-2 pl-9 pr-3 text-sm outline-none ring-slate-300 focus:ring-2 dark:border-slate-700 dark:bg-slate-950"
                  />
                </div>
                <Button onClick={() => setDark((d) => !d)} className="rounded-2xl" variant="outline">
                  <Moon className="mr-2 h-4 w-4" /> Theme
                </Button>
              </div>
            </header>

            <section className="mb-4 flex gap-2 overflow-x-auto pb-1">
              {decks.map((deck) => (
                <button
                  key={deck.id}
                  onClick={() => setActiveDeck(deck.id)}
                  className={`whitespace-nowrap rounded-2xl border px-4 py-2 text-sm font-bold transition ${
                    activeDeck === deck.id
                      ? "border-slate-950 bg-slate-950 text-white dark:border-white dark:bg-white dark:text-slate-950"
                      : "border-slate-200 bg-white text-slate-700 hover:bg-slate-100 dark:border-slate-800 dark:bg-slate-900 dark:text-slate-200 dark:hover:bg-slate-800"
                  }`}
                >
                  <span className="mr-2">{deck.emoji}</span>
                  {deck.name}
                </button>
              ))}
            </section>

            <AnimatePresence mode="wait">
              <motion.div
                key={mode}
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0, y: -10 }}
                transition={{ duration: 0.2 }}
              >
                {mode === "dashboard" && <Dashboard cards={filteredCards} dueCards={dueCards} weakCards={weakCards} avgMastery={avgMastery} history={history} streak={streak} setMode={setMode} />}
                {mode === "study" && <StudyMode cards={dueCards.length ? dueCards : filteredCards} onResult={updateCardResult} />}
                {mode === "test" && <TestMode cards={filteredCards} onResult={updateCardResult} />}
                {mode === "tutor" && <AITutor cards={filteredCards} weakCards={weakCards} />}
                {mode === "notes" && <NotesImporter onImport={importFromNotes} />}
                {mode === "focus" && <FocusRoom streak={streak} setStreak={setStreak} />}
                {mode === "library" && <Library cards={filteredCards} addCard={addCard} deleteCard={deleteCard} />}
              </motion.div>
            </AnimatePresence>
          </main>
        </div>
      </div>
    </div>
  );
}

function StudyMode({ cards, onResult }) {
  const [index, setIndex] = useState(0);
  const [flipped, setFlipped] = useState(false);
  const [typed, setTyped] = useState("");
  const card = cards[index % Math.max(cards.length, 1)];

  if (!cards.length) {
    return <Empty title="No cards to study" text="Add cards or switch to another deck." />;
  }

  function grade(quality) {
    onResult(card.id, quality);
    setFlipped(false);
    setTyped("");
    setIndex((i) => (i + 1) % cards.length);
  }

  return (
    <div className="grid gap-4 lg:grid-cols-[1fr_360px]">
      <Card className="min-h-[520px] rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
        <CardContent className="flex h-full flex-col p-6">
          <div className="mb-6 flex items-center justify-between">
            <div>
              <div className="text-sm font-bold text-slate-500">Smart Study</div>
              <h3 className="text-2xl font-black">{card.subject}</h3>
            </div>
            <div className="rounded-2xl bg-slate-100 px-3 py-2 text-sm font-bold dark:bg-slate-800">
              {index + 1} / {cards.length}
            </div>
          </div>

          <button onClick={() => setFlipped((f) => !f)} className="flex flex-1 items-center justify-center rounded-3xl border border-dashed border-slate-300 bg-slate-50 p-8 text-center transition hover:bg-slate-100 dark:border-slate-700 dark:bg-slate-950 dark:hover:bg-slate-800">
            <AnimatePresence mode="wait">
              {!flipped ? (
                <motion.div key="front" initial={{ opacity: 0, rotateX: -10 }} animate={{ opacity: 1, rotateX: 0 }} exit={{ opacity: 0, rotateX: 10 }}>
                  <p className="mb-4 text-sm font-bold uppercase tracking-widest text-slate-400">Prompt</p>
                  <h2 className="text-4xl font-black tracking-tight">{card.term}</h2>
                  <p className="mt-6 text-slate-500">Tap to reveal answer</p>
                </motion.div>
              ) : (
                <motion.div key="back" initial={{ opacity: 0, rotateX: -10 }} animate={{ opacity: 1, rotateX: 0 }} exit={{ opacity: 0, rotateX: 10 }}>
                  <p className="mb-4 text-sm font-bold uppercase tracking-widest text-slate-400">Answer</p>
                  <h2 className="text-2xl font-black leading-snug md:text-3xl">{card.answer}</h2>
                </motion.div>
              )}
            </AnimatePresence>
          </button>

          <div className="mt-5 grid gap-2 md:grid-cols-4">
            <Button onClick={() => grade("again")} variant="outline" className="rounded-2xl">Again</Button>
            <Button onClick={() => grade("hard")} variant="outline" className="rounded-2xl">Hard</Button>
            <Button onClick={() => grade("good")} className="rounded-2xl">Good</Button>
            <Button onClick={() => grade("easy")} className="rounded-2xl">Easy</Button>
          </div>
        </CardContent>
      </Card>

      <div className="space-y-4">
        <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
          <CardContent className="p-5">
            <div className="mb-3 flex items-center gap-2 font-black"><Highlighter className="h-5 w-5" /> Feynman Mode</div>
            <p className="mb-3 text-sm text-slate-500">Explain the card in your own words before flipping.</p>
            <textarea value={typed} onChange={(e) => setTyped(e.target.value)} placeholder="Type your simple explanation here..." className="h-36 w-full resize-none rounded-2xl border border-slate-200 bg-white p-3 text-sm outline-none focus:ring-2 dark:border-slate-700 dark:bg-slate-950" />
          </CardContent>
        </Card>
        <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
          <CardContent className="p-5">
            <div className="mb-3 flex items-center justify-between">
              <div className="font-black">Mastery</div>
              <div className="font-black">{card.mastery}%</div>
            </div>
            <div className="h-3 overflow-hidden rounded-full bg-slate-200 dark:bg-slate-800">
              <div className="h-full rounded-full bg-slate-950 dark:bg-white" style={{ width: `${card.mastery}%` }} />
            </div>
            <p className="mt-3 text-sm text-slate-500">Last result: {card.lastResult}</p>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}

function TestMode({ cards, onResult }) {
  const [started, setStarted] = useState(false);
  const [answers, setAnswers] = useState({});
  const quizCards = useMemo(() => [...cards].sort(() => Math.random() - 0.5).slice(0, Math.min(8, cards.length)), [cards, started]);
  const score = quizCards.reduce((sum, card) => sum + (answers[card.id]?.toLowerCase().trim() === card.answer.toLowerCase().trim() ? 1 : 0), 0);
  const completed = quizCards.length && Object.keys(answers).length === quizCards.length;

  if (!cards.length) return <Empty title="No cards available" text="Add cards first to generate a test." />;

  return (
    <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
      <CardContent className="p-6">
        <div className="mb-6 flex flex-col gap-3 md:flex-row md:items-center md:justify-between">
          <div>
            <h3 className="text-2xl font-black">Test Generator</h3>
            <p className="text-sm text-slate-500">Creates a written-response quiz from your current deck.</p>
          </div>
          <Button onClick={() => { setStarted(true); setAnswers({}); }} className="rounded-2xl">
            <GraduationCap className="mr-2 h-4 w-4" /> Generate New Test
          </Button>
        </div>

        {!started ? (
          <div className="rounded-3xl bg-slate-50 p-8 text-center dark:bg-slate-950">
            <Trophy className="mx-auto mb-4 h-12 w-12" />
            <h4 className="text-xl font-black">Ready for exam mode?</h4>
            <p className="mx-auto mt-2 max-w-xl text-slate-500">This mode is harder than Quizlet-style matching because it makes you produce the answer from memory.</p>
          </div>
        ) : (
          <div className="space-y-4">
            {quizCards.map((card, i) => (
              <div key={card.id} className="rounded-3xl border border-slate-200 p-4 dark:border-slate-800">
                <div className="mb-2 text-sm font-bold text-slate-400">Question {i + 1}</div>
                <div className="mb-3 text-lg font-black">{card.term}</div>
                <input value={answers[card.id] || ""} onChange={(e) => setAnswers((prev) => ({ ...prev, [card.id]: e.target.value }))} placeholder="Type exact answer..." className="w-full rounded-2xl border border-slate-200 bg-white p-3 outline-none focus:ring-2 dark:border-slate-700 dark:bg-slate-950" />
                {completed && (
                  <div className="mt-3 rounded-2xl bg-slate-50 p-3 text-sm dark:bg-slate-950">
                    <b>Correct answer:</b> {card.answer}
                  </div>
                )}
              </div>
            ))}
            {completed && (
              <div className="rounded-3xl bg-slate-950 p-5 text-white dark:bg-white dark:text-slate-950">
                <div className="text-2xl font-black">Score: {score} / {quizCards.length}</div>
                <p className="mt-1 opacity-80">Cards answered exactly are marked stronger. Review the others again.</p>
                <div className="mt-4 flex flex-wrap gap-2">
                  {quizCards.forEach((card) => onResult(card.id, answers[card.id]?.toLowerCase().trim() === card.answer.toLowerCase().trim() ? "good" : "again"))}
                </div>
              </div>
            )}
          </div>
        )}
      </CardContent>
    </Card>
  );
}

function AITutor({ cards, weakCards }) {
  const [messages, setMessages] = useState([
    {
      role: "tutor",
      text: "Hi! I’m your AI Tutor. Pick a card or ask me something like: ‘explain this simply,’ ‘quiz me,’ ‘give me a hint,’ or ‘make me a study plan.’",
    },
  ]);
  const [input, setInput] = useState("");
  const [selectedId, setSelectedId] = useState(cards[0]?.id || "");
  const selectedCard = cards.find((c) => c.id === selectedId) || cards[0];

  function localTutorReply(question) {
    const q = question.toLowerCase();
    const card = selectedCard;
    const weakest = weakCards?.[0];

    if (!card) {
      return "Add a few cards first, then I can tutor you on them, quiz you, and build a study plan.";
    }

    if (q.includes("hint")) {
      return `Hint: for “${card.term},” focus on the main idea first. Try saying what it does, why it matters, and one example. I won’t give the full answer yet.`;
    }

    if (q.includes("quiz") || q.includes("test")) {
      return `Mini-quiz: Without looking, explain “${card.term}” in one sentence. Then add one example. After that, compare your response with: ${card.answer}`;
    }

    if (q.includes("simple") || q.includes("explain") || q.includes("eli5")) {
      return `Simple explanation: “${card.term}” means ${card.answer} Think of it as the core idea you need to remember. A good way to study it is to explain it out loud, then check if you included the key words.`;
    }

    if (q.includes("wrong") || q.includes("mistake") || q.includes("confused")) {
      return `Let’s debug the confusion. For “${card.term},” students usually miss either the definition, the purpose, or the example. The correct idea is: ${card.answer}. Try rewriting it in your own words, then ask me for a hint if you get stuck.`;
    }

    if (q.includes("plan") || q.includes("schedule")) {
      return `Personal study plan: 1) Spend 5 minutes reviewing weak cards. 2) Practice “${weakest?.term || card.term}” first because it looks weakest. 3) Do 10 active-recall cards. 4) Take a short generated test. 5) Re-study anything below 70% mastery.`;
    }

    return `For “${card.term},” here’s how I’d tutor it: first understand the meaning, then say it from memory, then use an example. The answer is: ${card.answer}. Now try explaining it in your own words and I can help improve it.`;
  }

  function sendMessage(preset) {
    const text = preset || input.trim();
    if (!text) return;
    const reply = localTutorReply(text);
    setMessages((prev) => [...prev, { role: "student", text }, { role: "tutor", text: reply }]);
    setInput("");
  }

  return (
    <div className="grid gap-4 lg:grid-cols-[1fr_360px]">
      <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
        <CardContent className="flex h-[680px] flex-col p-5">
          <div className="mb-4 flex items-center gap-3">
            <div className="rounded-2xl bg-slate-100 p-3 dark:bg-slate-800"><MessageCircle className="h-6 w-6" /></div>
            <div>
              <h3 className="text-2xl font-black">AI Tutor</h3>
              <p className="text-sm text-slate-500">Hints, explanations, mini-quizzes, mistake help, and study plans.</p>
            </div>
          </div>

          <div className="mb-4 flex flex-wrap gap-2">
            <Button onClick={() => sendMessage("Give me a hint")} variant="outline" className="rounded-2xl">Hint</Button>
            <Button onClick={() => sendMessage("Explain this simply")} variant="outline" className="rounded-2xl">Explain Simply</Button>
            <Button onClick={() => sendMessage("Quiz me")} variant="outline" className="rounded-2xl">Quiz Me</Button>
            <Button onClick={() => sendMessage("Make me a study plan")} variant="outline" className="rounded-2xl">Study Plan</Button>
          </div>

          <div className="flex-1 space-y-3 overflow-y-auto rounded-3xl bg-slate-50 p-4 dark:bg-slate-950">
            {messages.map((message, index) => (
              <div key={index} className={`flex ${message.role === "student" ? "justify-end" : "justify-start"}`}>
                <div className={`max-w-[82%] rounded-3xl px-4 py-3 text-sm leading-relaxed ${message.role === "student" ? "bg-slate-950 text-white dark:bg-white dark:text-slate-950" : "bg-white text-slate-700 shadow-sm dark:bg-slate-900 dark:text-slate-200"}`}>
                  <div className="mb-1 text-xs font-black uppercase tracking-widest opacity-50">{message.role === "student" ? "You" : "Tutor"}</div>
                  {message.text}
                </div>
              </div>
            ))}
          </div>

          <div className="mt-4 flex gap-2">
            <input
              value={input}
              onChange={(e) => setInput(e.target.value)}
              onKeyDown={(e) => e.key === "Enter" && sendMessage()}
              placeholder="Ask the tutor anything..."
              className="flex-1 rounded-2xl border border-slate-200 bg-white p-3 outline-none focus:ring-2 dark:border-slate-700 dark:bg-slate-950"
            />
            <Button onClick={() => sendMessage()} className="rounded-2xl px-6">Send</Button>
          </div>
        </CardContent>
      </Card>

      <div className="space-y-4">
        <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
          <CardContent className="p-5">
            <h4 className="mb-3 text-xl font-black">Current Topic</h4>
            <select value={selectedCard?.id || ""} onChange={(e) => setSelectedId(e.target.value)} className="mb-4 w-full rounded-2xl border border-slate-200 bg-white p-3 outline-none dark:border-slate-700 dark:bg-slate-950">
              {cards.map((card) => <option key={card.id} value={card.id}>{card.term}</option>)}
            </select>
            {selectedCard ? (
              <div className="rounded-3xl bg-slate-50 p-4 dark:bg-slate-950">
                <div className="text-xs font-bold uppercase tracking-widest text-slate-400">{selectedCard.subject}</div>
                <div className="mt-1 text-lg font-black">{selectedCard.term}</div>
                <p className="mt-2 text-sm text-slate-500">{selectedCard.answer}</p>
                <div className="mt-4 h-2 overflow-hidden rounded-full bg-slate-200 dark:bg-slate-800">
                  <div className="h-full rounded-full bg-slate-950 dark:bg-white" style={{ width: `${selectedCard.mastery}%` }} />
                </div>
              </div>
            ) : <p className="text-sm text-slate-500">No cards yet.</p>}
          </CardContent>
        </Card>

        <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
          <CardContent className="p-5">
            <div className="mb-2 flex items-center gap-2 font-black"><Settings className="h-5 w-5" /> Real AI Upgrade</div>
            <p className="text-sm text-slate-500">This version uses a local tutor brain so it works immediately. To make it a real AI tutor, connect the Send button to a backend endpoint like <b>/api/tutor</b>, then call OpenAI, Claude, or Gemini from the server.</p>
          </CardContent>
        </Card>
      </div>
    </div>
  );
}

function FocusRoom({ streak, setStreak }) {
  const [seconds, setSeconds] = useState(25 * 60);
  const [running, setRunning] = useState(false);
  const interval = useRef(null);

  useEffect(() => {
    if (!running) return;
    interval.current = setInterval(() => {
      setSeconds((s) => {
        if (s <= 1) {
          clearInterval(interval.current);
          setRunning(false);
          setStreak((x) => x + 1);
          return 25 * 60;
        }
        return s - 1;
      });
    }, 1000);
    return () => clearInterval(interval.current);
  }, [running, setStreak]);

  const minutes = Math.floor(seconds / 60).toString().padStart(2, "0");
  const secs = (seconds % 60).toString().padStart(2, "0");

  return (
    <div className="grid gap-4 lg:grid-cols-[1fr_360px]">
      <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
        <CardContent className="p-8 text-center">
          <Timer className="mx-auto mb-4 h-12 w-12" />
          <h3 className="text-2xl font-black">Focus Room</h3>
          <p className="mx-auto mt-2 max-w-xl text-slate-500">A built-in Pomodoro timer so studying and reviewing stay in one place.</p>
          <div className="my-10 text-7xl font-black tracking-tight md:text-8xl">{minutes}:{secs}</div>
          <div className="flex justify-center gap-2">
            <Button onClick={() => setRunning((r) => !r)} className="rounded-2xl px-8">{running ? "Pause" : "Start"}</Button>
            <Button onClick={() => { setRunning(false); setSeconds(25 * 60); }} variant="outline" className="rounded-2xl px-8">Reset</Button>
          </div>
        </CardContent>
      </Card>
      <Card className="rounded-3xl border-slate-200 bg-white shadow-sm dark:border-slate-800 dark:bg-slate-900">
        <CardContent className="p-5">
          <h4 className="mb-3 text-xl font-black">Quest Board</h4>
          {[
            ["Complete one focus session", seconds < 25 * 60],
            ["Review 10 cards", false],
            ["Fix 3 weak spots", false],
            [`Maintain ${streak}-day streak`, true],
          ].map(([label, done]) => (
            <div key={label} className="mb-3 flex items-center justify-between rounded-2xl bg-slate-50 p-3 dark:bg-slate-950">
              <span className="font-semibold">{label}</span>
              {done ? <CheckCircle2 className="h-5 w-5" /> : <div className="h-5 w-5 rounded-full border border-slate-300" />}
            </div>
          ))}
        </CardContent>
      </Card>
    </div>
  );
}
